/**-------------------------------------------------
 - Raspberry Pi起動時にやっておいた方がいいこと
 -------------------------------------------------*/

 このreadme.txtがあるカレントディレクトリにて下記のコマンド実行

 make update
 make install

/**-------------------------------------------------
 - spi/i2cの有効可
 -------------------------------------------------*/
 sudo raspi-config
 Advanced Optionsから設定、全てEnabledに変更

 ssh
 spi
 i2c
 camera
 serial

 再起動後、lsmodコマンド実行
 spi_bcm2708---などが表示される

 /**-------------------------------------------------
  - ssh connect
  -------------------------------------------------*/
 ssh -X pi@raspberrypi.local
 raspberry

 /**-------------------------------------------------
  - ssh unconnect
  -------------------------------------------------*/
  exit
  sudo shutdown -h now

 /**-------------------------------------------------
  - ipconfig 固定
  -------------------------------------------------*/
  ifconfig
  sudo vim /etc/dhcpcd.conf


 /**-------------------------------------------------
  - ファイルディレクトリやりとり
  -------------------------------------------------*/
  1.ラズパイに送信
  scp -r ディレクトリ名 pi@192.168.10.7:/home/pi/wprkspace
  scp ファイル名 pi@192.168.10.7:/home/pi/wprkspace

  2.ラズパイから取得
  scp -r pi@192.168.10.7:/home/pi/wprkspace/ディレクトリ名
  scp pi@192.168.10.7:/home/pi/wprkspace/ファイル名


 /**-------------------------------------------------
 - gpio確認
 -------------------------------------------------*/
 gpio readall
